PREFIX foaf: <http://xmlns.com/foaf/0.1/>

PREFIX owl: <http://www.w3.org/2002/07/owl#>

CONSTRUCT {
   ?creatorAgentiri rdf:type foaf:Agent ;
      foaf:name ?creatorAgentName.
  
} WHERE {

?c mydata:creatorAgentID ?creatorAgentID . 
?c mydata:creatorAgentName ?creatorAgentNameprep.   
bind(STRLANG(?creatorAgentNameprep, "en") AS ?creatorAgentName)
      
bind(iri(concat("http://dskg.org/entity/agent/", spif:encodeURL(?creatorAgentID))) as ?creatorAgentiri) 
  
 } LIMIT 100
